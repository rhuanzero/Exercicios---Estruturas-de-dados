2) Para a classe HeapBinariaMinima, crie um construtor que receba uma Heap Máxima
como parâmetro e a converta em Heap Mínima. Esse construtor deve possuir o protótipo
abaixo:
public HeapBinariaMinima(HeapBinariaMaxima heapMaxima);

Métodos adicionados em HeapBinariaMaxima:

public int getTam() {
	return tam;
}
public int get(int i) {
	return vetor[i+1];
}
public int getN() {
	return n;
}

public HeapBinariaMinima(HeapBinariaMaxima heapMaxima) {
		this.n = heapMaxima.getN();		
		this.tam = heapMaxima.getTam();
		this.vetor = new int[tam + 1];
	
		
		for(int i = 0;i<n;i++) 
			vetor[i+1] = heapMaxima.get(i);
	
		int ultno = n/2;
		
		for( int i = ultno; i > 0; i-- ) {
			int x = vetor[i];
			
			while(2*i <= n) { // Enquanto o filho esquerdo existir
				int filhoesq = 2*i, filhodir = 2*i+1, menorfilho;
				
				menorfilho = filhoesq;
				
				if(filhodir<=n) { // Verifica se o filho existe
					if(vetor[menorfilho] > vetor[filhodir]) { // Verifica se o filho esquerdo é o menorfilho
						menorfilho = filhodir;
					}
				}
				if(vetor[menorfilho] < x) {
					vetor[i] = vetor[menorfilho];
				}
				else
					break;
				
				i = menorfilho;
			}
		
			vetor[i] = x;
		
		}
		
	}

Complexidade do construtor :
O(n) - sendo n o numero de elementos na heap
